$("#setDateTest").datepicker('setDate', "2019-12-07"); // 날짜 설정
$("#setDateTest").datepicker('option', 'minDate', "2011-12-07"); // 최소날짜 설정
$("#setDateTest").datepicker('option', 'maxDate', "2011-12-07"); // 최대날짜 설정
$("#setDateTest").datepicker("option", "disabled", true); // datepicker disable
$("#setDateTest").datepicker("option", "disabled", false); // datepicker enabled
